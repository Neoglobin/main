def sort(tp,val):
    try:
        first_index = tp.index(val)
    except ValueError:
        return ()

    try:
        second_index = tp.index(val, first_index + 1)
        return tp[first_index:second_index + 1]
    except ValueError:
        return tp[first_index:]
